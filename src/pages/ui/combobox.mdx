# Combobox

Autocomplete input and command palette with a list of suggestions.

## Installation

### CLI

```package-dlx
shadcn@latest add combobox
```

### Manual

Copy and paste the following code into your project.

```tsx file=../../registry/ui/combobox.tsx showLineNumbers

```

## Usage

```tsx
import {
  Combobox,
  ComboboxContent,
  ComboboxControl,
  ComboboxInput,
  ComboboxItem,
  ComboboxTrigger,
} from "~/components/ui/combobox";
```

```tsx showLineNumbers
const frameworks = ["Next.js", "SvelteKit", "Nuxt.js", "Remix", "Astro"];

<Combobox
  options={frameworks}
  placeholder="Select a framework..."
  itemComponent={(props) => (
    <ComboboxItem item={props.item}>{props.item.rawValue}</ComboboxItem>
  )}
>
  <ComboboxControl>
    <ComboboxInput />
    <ComboboxTrigger />
  </ComboboxControl>
  <ComboboxContent />
</Combobox>
```
